Event Example
Click addEventListener('click', function() { alert('You clicked me'); });
Mouseover addEventListener('mouseover', function() { alert('You hovered over me'); });
Mouseout addEventListener('mouseout', function() { alert('You hovered out of me'); });
Keydown addEventListener('keydown', function() { alert('You pressed a key'); });
Keyup addEventListener('keyup', function() { alert('You released a key'); });
Load addEventListener('load', function() { alert('The page has loaded'); });
Focus addEventListener('focus', function() { alert('You focused on me'); });
Blur addEventListener('blur', function() { alert('You blurred out of me'); });
Submit addEventListener('submit', function() { alert('You submitted the form'); });
Change addEventListener('change', function() { alert('You changed the input'); });
Resize addEventListener('resize', function() { alert('You resized the window'); });
Scroll addEventListener('scroll', function() { alert('You scrolled the page'); });
Contextmenu addEventListener('contextmenu', function() { alert('You right-clicked'); });
Dblclick addEventListener('dblclick', function() { alert('You double-clicked'); });
Mousedown addEventListener('mousedown', function() { alert('You pressed the mouse button'); });
Mouseup addEventListener('mouseup', function() { alert('You released the mouse button'); });
Mousemove addEventListener('mousemove', function() { alert('You moved the mouse'); });
